import { Router } from "express";

const router = Router();

router.get("/hackernews", async (req, res) => {
  const query = encodeURIComponent((req.query.query as string) || "security vulnerability exploit");
  try {
    const url = `https://hn.algolia.com/api/v1/search?query=${query}&tags=story&hitsPerPage=30`;
    const upstream = await fetch(url, {
      headers: { "User-Agent": "UnifiedThreatAlert/1.0" },
      signal: AbortSignal.timeout(10_000),
    });
    if (!upstream.ok) {
      res.status(502).json({ error: "upstream_error" });
      return;
    }
    const data = await upstream.json() as { hits: unknown[] };
    res.json({ hits: data.hits ?? [] });
  } catch (err: any) {
    res.status(502).json({ error: "fetch_failed", message: err?.message });
  }
});

export default router;
EOF
